CARGS hdc1 : mandelbrot
# 0 i !
 BEGIN
  i @ # 3 e * <
 WHILE
  # 0 j !
  BEGIN
   j @ # 2 e * <
  WHILE
   i @ # 2 e * - a !
   j @ e - b !
   # 0 x !
   # 0 y !
   # 0 t !
   # 1 cont !

   BEGIN
    t @ # 20 < cont @ &&
   WHILE
    x @ x @ * e / y @ y @ * e / - a @ + x1 !
    # 2 x @ * y @ * e / b @ + y !
    x1 @ x !
    
    x @ x @ * e / y @ y @ * e / + # 4 e * >
    IF # 0 cont ! ELSE nop THEN

    t @ # 1 + t !
   REPEAT

   t @ # FF * # 8 / r !
   r @ # FF > IF # FF r ! ELSE nop THEN
   x @ x @ * y @ y @ * + # 10 * e / e / g !
   g @ # FF > IF # FF g ! ELSE nop THEN
   x @ x @ * y @ y @ * + # 400 * e / e / bl !
   bl @ # FF > IF # FF bl ! ELSE nop THEN

   r @ g @ bl @ RGB c !

   c @ i @ j @ # 32 + COORD hdc1 SETPIXEL DROP
   j @ # 1 + j !
  REPEAT
  i @ # 1 + i !
 REPEAT
END ;
